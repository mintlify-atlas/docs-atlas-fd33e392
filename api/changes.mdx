## change

Modify an Automerge document within a change function.

```typescript
function change<T>(
  doc: Doc<T>,
  options: string | ChangeOptions<T> | ChangeFn<T>,
  callback?: ChangeFn<T>
): Doc<T>
```

<ParamField path="doc" type="Doc<T>" required>
  The document to modify
</ParamField>

<ParamField path="options" type="string | ChangeOptions<T> | ChangeFn<T>" required>
  Either a message string, options object, or the change function itself
  
  <Expandable title="ChangeOptions<T>">
    <ParamField path="message" type="string">
      A message describing the changes
    </ParamField>
    <ParamField path="time" type="number | undefined">
      Unix timestamp (in seconds) for the change. Defaults to current time. Set to `undefined` for no timestamp.
    </ParamField>
    <ParamField path="patchCallback" type="PatchCallback<T>">
      Callback to receive patches generated by this change
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="callback" type="ChangeFn<T>">
  The change function (if `options` is not a function)
</ParamField>

<ResponseField name="return" type="Doc<T>">
  A new document with the changes applied
</ResponseField>

### Examples

<CodeGroup>

```typescript Basic Change
let doc = Automerge.init()

doc = Automerge.change(doc, d => {
  d.title = "Hello World"
  d.count = 0
})
```

```typescript With Message
doc = Automerge.change(doc, "Update counter", d => {
  d.count++
})
```

```typescript With Options
doc = Automerge.change(doc, {
  message: "Add items",
  time: Math.floor(Date.now() / 1000)
}, d => {
  d.items = ["apple", "banana"]
})
```

```typescript With Patch Callback
doc = Automerge.change(doc, {
  message: "Modify data",
  patchCallback: (patches, info) => {
    console.log('Changes made:', patches)
    console.log('Before:', info.before)
    console.log('After:', info.after)
  }
}, d => {
  d.data = "updated"
})
```

```typescript Nested Changes
type TodoList = {
  todos: Array<{ text: string; done: boolean }>
}

let doc = Automerge.init<TodoList>()

doc = Automerge.change(doc, d => {
  d.todos = []
  d.todos.push({ text: "Buy milk", done: false })
  d.todos.push({ text: "Write docs", done: true })
  d.todos[0].done = true
})
```

</CodeGroup>

## emptyChange

Make a change that doesn't modify the document.

```typescript
function emptyChange<T>(
  doc: Doc<T>,
  options?: string | ChangeOptions<T>
): Doc<T>
```

<ParamField path="doc" type="Doc<T>" required>
  The document to add an empty change to
</ParamField>

<ParamField path="options" type="string | ChangeOptions<T>">
  Either a message or change options
</ParamField>

<ResponseField name="return" type="Doc<T>">
  The document with an empty change recorded
</ResponseField>

### Examples

<CodeGroup>

```typescript Sign Merged Changes
let doc1 = Automerge.from({ data: "original" })
let doc2 = Automerge.clone(doc1)

doc2 = Automerge.change(doc2, d => d.data = "modified")

// Merge remote changes
doc1 = Automerge.merge(doc1, doc2)

// Create empty change to sign/timestamp the merge
doc1 = Automerge.emptyChange(doc1, "Merged changes from peer")
```

```typescript With Timestamp
doc = Automerge.emptyChange(doc, {
  message: "Checkpoint",
  time: Math.floor(Date.now() / 1000)
})
```

</CodeGroup>

<Note>
  Empty changes are useful for creating dependencies on merged changes, allowing you to sign or timestamp a merge operation.
</Note>

## getChanges

Get changes that are in one document but not another.

```typescript
function getChanges<T>(oldState: Doc<T>, newState: Doc<T>): Change[]
```

<ParamField path="oldState" type="Doc<T>" required>
  The older document state
</ParamField>

<ParamField path="newState" type="Doc<T>" required>
  The newer document state
</ParamField>

<ResponseField name="return" type="Change[]">
  Array of binary-encoded changes present in `newState` but not in `oldState`
</ResponseField>

### Examples

<CodeGroup>

```typescript Get Changes
let doc1 = Automerge.from({ counter: 0 })
const snapshot = Automerge.clone(doc1)

doc1 = Automerge.change(doc1, d => d.counter++)
doc1 = Automerge.change(doc1, d => d.counter++)

const changes = Automerge.getChanges(snapshot, doc1)
console.log(`Made ${changes.length} changes`) // 2
```

```typescript Send Changes to Peer
const beforeHeads = Automerge.getHeads(doc)

// Make some changes
doc = Automerge.change(doc, d => d.value = "new")

// Get just the new changes
const before = Automerge.view(doc, beforeHeads)
const changes = Automerge.getChanges(before, doc)

// Send changes to peer
network.send(changes)
```

</CodeGroup>

<Warning>
  This will crash if there are changes in `oldState` that are not in `newState`. Ensure `newState` contains all changes from `oldState`.
</Warning>

## getAllChanges

Get all changes in a document.

```typescript
function getAllChanges<T>(doc: Doc<T>): Change[]
```

<ParamField path="doc" type="Doc<T>" required>
  The document to get changes from
</ParamField>

<ResponseField name="return" type="Change[]">
  Array of all binary-encoded changes in the document
</ResponseField>

### Examples

```typescript
let doc = Automerge.from({ title: "Doc" })
doc = Automerge.change(doc, d => d.title = "Document")
doc = Automerge.change(doc, d => d.author = "Alice")

const allChanges = Automerge.getAllChanges(doc)
console.log(`Document has ${allChanges.length} changes`)

// Can recreate document from changes
let doc2 = Automerge.init()
;[doc2] = Automerge.applyChanges(doc2, allChanges)
```

## applyChanges

Apply changes received from another document.

```typescript
function applyChanges<T>(
  doc: Doc<T>,
  changes: Change[],
  opts?: ApplyOptions<T>
): [Doc<T>]
```

<ParamField path="doc" type="Doc<T>" required>
  The document to apply changes to
</ParamField>

<ParamField path="changes" type="Change[]" required>
  Array of binary-encoded changes to apply
</ParamField>

<ParamField path="opts" type="ApplyOptions<T>">
  <Expandable title="ApplyOptions<T>">
    <ParamField path="patchCallback" type="PatchCallback<T>">
      Callback to receive patches as changes are applied
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="return" type="[Doc<T>]">
  Array with a single element: the updated document
</ResponseField>

### Examples

<CodeGroup>

```typescript Apply Changes
let doc1 = Automerge.from({ items: [] })
let doc2 = Automerge.clone(doc1)

// Make changes on doc1
doc1 = Automerge.change(doc1, d => {
  d.items.push("apple")
})

// Get the changes and apply to doc2
const changes = Automerge.getChanges(doc2, doc1)
;[doc2] = Automerge.applyChanges(doc2, changes)

console.log(doc2.items) // ["apple"]
```

```typescript With Patch Callback
const patchCallback = (patches, info) => {
  patches.forEach(patch => {
    console.log('Patch:', patch.action, 'at', patch.path)
  })
}

;[doc] = Automerge.applyChanges(doc, changes, { patchCallback })
```

```typescript Rebuild from All Changes
const allChanges = Automerge.getAllChanges(originalDoc)

let newDoc = Automerge.init()
;[newDoc] = Automerge.applyChanges(newDoc, allChanges)

// newDoc now equals originalDoc
```

</CodeGroup>

<Note>
  Changes with missing dependencies are stored and automatically applied when those dependencies arrive. The return value is an array for legacy reasons.
</Note>

## loadIncremental

Load incremental changes into a document.

```typescript
function loadIncremental<T>(
  doc: Doc<T>,
  data: Uint8Array,
  opts?: ApplyOptions<T>
): Doc<T>
```

<ParamField path="doc" type="Doc<T>" required>
  The document to load changes into
</ParamField>

<ParamField path="data" type="Uint8Array" required>
  Binary data from `saveIncremental()`, `save()`, or `getLastLocalChange()`
</ParamField>

<ParamField path="opts" type="ApplyOptions<T>">
  Options including patch callback
</ParamField>

<ResponseField name="return" type="Doc<T>">
  The updated document
</ResponseField>

### Examples

<CodeGroup>

```typescript Incremental Sync
// Peer 1: Make a change and save incrementally
let doc1 = Automerge.from({ counter: 0 })
doc1 = Automerge.change(doc1, d => d.counter++)
const incrementalData = Automerge.saveIncremental(doc1)

// Peer 2: Load the incremental data
let doc2 = Automerge.from({ counter: 0 })
doc2 = Automerge.loadIncremental(doc2, incrementalData)

console.log(doc2.counter) // 1
```

```typescript Stream Changes
// Initial document
let doc = Automerge.init()

// Receive and apply changes as they arrive
network.onData((chunk: Uint8Array) => {
  doc = Automerge.loadIncremental(doc, chunk, {
    patchCallback: (patches) => {
      updateUI(patches)
    }
  })
})
```

</CodeGroup>

## saveIncremental

Create binary data for incremental saves.

```typescript
function saveIncremental<T>(doc: Doc<T>): Uint8Array
```

<ParamField path="doc" type="Doc<T>" required>
  The document to save incrementally
</ParamField>

<ResponseField name="return" type="Uint8Array">
  Binary data containing changes since the last incremental save
</ResponseField>

### Examples

<CodeGroup>

```typescript Append to File
let doc = Automerge.from({ logs: [] })

// Save full document
fs.writeFileSync('doc.automerge', Automerge.save(doc))

// Make changes and append incremental saves
doc = Automerge.change(doc, d => d.logs.push("entry 1"))
fs.appendFileSync('doc.automerge', Automerge.saveIncremental(doc))

doc = Automerge.change(doc, d => d.logs.push("entry 2"))
fs.appendFileSync('doc.automerge', Automerge.saveIncremental(doc))
```

```typescript Real-time Sync
let doc = Automerge.from({ data: "initial" })

// After each change, send incremental data to peers
doc = Automerge.change(doc, d => d.data = "updated")
const delta = Automerge.saveIncremental(doc)
broadcast(delta)
```

</CodeGroup>

## getLastLocalChange

Get the binary representation of the last change made to this document.

```typescript
function getLastLocalChange<T>(doc: Doc<T>): Change | undefined
```

<ParamField path="doc" type="Doc<T>" required>
  The document to get the last change from
</ParamField>

<ResponseField name="return" type="Change | undefined">
  Binary-encoded last change, or undefined if no local changes have been made
</ResponseField>

### Examples

```typescript
let doc = Automerge.from({ value: 1 })

doc = Automerge.change(doc, d => d.value = 2)
const lastChange = Automerge.getLastLocalChange(doc)

if (lastChange) {
  // Send to other peers immediately
  broadcast(lastChange)
}
```

## getHistory

Get the complete history of a document as an array of states, one per change.

```typescript
function getHistory<T>(doc: Doc<T>): State<T>[]
```

<ParamField path="doc" type="Doc<T>" required>
  The document
</ParamField>

**Returns**: `State<T>[]` - Array of document states, each containing `snapshot` and `change` metadata

### Example

```typescript
let doc = Automerge.init()
doc = Automerge.change(doc, d => { d.count = 1 })
doc = Automerge.change(doc, d => { d.count = 2 })

const history = Automerge.getHistory(doc)
console.log(history.length) // 2
console.log(history[0].snapshot) // { count: 1 }
console.log(history[1].snapshot) // { count: 2 }
```

## getConflicts

Get all conflicting values for a property when concurrent assignments occur.

```typescript
function getConflicts<T>(
  object: any,
  prop: Prop
): Conflicts | undefined
```

<ParamField path="object" type="any" required>
  The parent object containing the property (not the document root)
</ParamField>

<ParamField path="prop" type="Prop" required>
  The property name or index
</ParamField>

**Returns**: `Conflicts | undefined` - Object mapping operation IDs to values, or undefined if no conflicts

### Example

```typescript
let doc1 = Automerge.init()
doc1 = Automerge.change(doc1, d => { d.name = "Alice" })

let doc2 = Automerge.clone(doc1)
doc1 = Automerge.change(doc1, d => { d.name = "Alice Smith" })
doc2 = Automerge.change(doc2, d => { d.name = "Alice Jones" })

const merged = Automerge.merge(doc1, doc2)
console.log(merged.name) // One value (deterministic)

const conflicts = Automerge.getConflicts(merged, "name")
console.log(Object.values(conflicts)) // ["Alice Smith", "Alice Jones"]
```

<Warning>
  Pass the **parent object** containing the property, not the document root. For nested properties, pass the immediate parent.
</Warning>

## getMissingDeps

Get the change hashes that are referenced but not present in a document. Useful for sync to determine what changes need to be sent.

```typescript
function getMissingDeps<T>(doc: Doc<T>, heads: Heads): Heads
```

<ParamField path="doc" type="Doc<T>" required>
  The document
</ParamField>

<ParamField path="heads" type="Heads" required>
  The heads to check against
</ParamField>

**Returns**: `Heads` - Array of missing change hashes

### Example

```typescript
const missingChanges = Automerge.getMissingDeps(doc, remoteHeads)
if (missingChanges.length > 0) {
  // Send these changes to the peer
}
```
